async function performFullScan() { const btn = document.querySelector('.scan-btn'); const status = document.getElementById('scan-status'); const table = document.getElementById('scan-results-table'); const tbody = document.getElementById('scan-results-body'); btn.disabled = true; btn.textContent = '⏳ Memindai...'; status.textContent = 'Mengambil data...'; table.style.display = 'none'; tbody.innerHTML = ''; try { const start = Date.now(); const res = await fetch('/api/scanner-full'); const data = await res.json(); const elapsed = ((Date.now()-start)/1000).toFixed(1); if (!data.results?.length) { status.textContent = 'Tidak ada hasil.'; return; } data.results.forEach((coin, i) => { const row = tbody.insertRow(); const prob = coin.breakoutProbability || {}; const color = prob.score >=70 ? '#7bffb1' : prob.score >=50 ? '#ffb87b' : '#aaa'; // Pola Grafik let patternsHtml = '-'; if (coin.chartPatterns?.length) { patternsHtml = coin.chartPatterns.map(p => `${p.name}` ).join('
'); } // Elliott Wave let elliottHtml = '-'; if (coin.elliottWave && coin.elliottWave.wave !== '-' && coin.elliottWave.wave !== 'Data kurang') { const confColor = coin.elliottWave.confidence >= 60 ? '#7bffb1' : '#ffb87b'; elliottHtml = `${coin.elliottWave.wave}
${coin.elliottWave.description}`; } // SMC let smcHtml = '-'; if (coin.smc && coin.smc.signal !== 'Neutral') { const smcColor = coin.smc.signal === 'Bullish' ? '#7bffb1' : '#ff7b7b'; smcHtml = `${coin.smc.signal}
${coin.smc.summary}`; } row.innerHTML = ` ${i+1} ${coin.symbol}
${coin.name} $${coin.price?.toFixed(4) || 'N/A'} ${coin.priceChange24h?.toFixed(2) || '0'}% ${patternsHtml} ${elliottHtml} ${smcHtml} ${prob.score}%
${prob.interpretation} ${(prob.reasons||[]).join('
')} `; }); table.style.display = 'table'; status.textContent = `✅ ${data.totalScanned} koin dipindai dalam ${elapsed} detik.`; } catch (e) { status.textContent = '❌ Gagal.'; } finally { btn.disabled = false; btn.textContent = '🚀 Mulai Pemindaian Penuh'; } }
